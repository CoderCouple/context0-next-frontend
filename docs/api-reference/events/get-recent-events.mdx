## Overview

This endpoint provides the most recent events for the authenticated user, designed to help build context for AI interactions and user behavior analysis. It returns events in reverse chronological order.

## Query Parameters

<ParamField query="limit" type="integer" default="10">
  Number of recent events to return (max: 50)
</ParamField>

## Response

<ResponseField name="events" type="array">
  Array of recent event objects ordered by timestamp (newest first)
  <Expandable title="Event Object">
    <ResponseField name="event_id" type="string">
      Unique identifier for the event
    </ResponseField>
    <ResponseField name="event_type" type="string">
      Type of the event
    </ResponseField>
    <ResponseField name="user_id" type="string">
      ID of the user associated with the event
    </ResponseField>
    <ResponseField name="metadata" type="object">
      Additional event data and context
    </ResponseField>
    <ResponseField name="timestamp" type="string">
      When the event occurred (ISO format)
    </ResponseField>
    <ResponseField name="relevance_score" type="number">
      Contextual relevance score (0.0 to 1.0)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="context_summary" type="object">
  Summary of recent activity patterns
  <Expandable title="Context Summary">
    <ResponseField name="primary_activities" type="array">
      Most common event types in the recent period
    </ResponseField>
    <ResponseField name="active_sessions" type="integer">
      Number of active user sessions
    </ResponseField>
    <ResponseField name="time_range" type="object">
      Time range covered by the events
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://api.contextzero.ai/api/v1/event/context?limit=15" \
  -H "Authorization: Bearer YOUR_API_TOKEN" \
  -H "Content-Type: application/json"
```

```javascript JavaScript
const response = await fetch('https://api.contextzero.ai/api/v1/event/context?limit=15', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_API_TOKEN',
    'Content-Type': 'application/json',
  },
});

const recentEvents = await response.json();
```

```python Python
import requests

response = requests.get(
    "https://api.contextzero.ai/api/v1/event/context",
    headers={
        "Authorization": "Bearer YOUR_API_TOKEN",
        "Content-Type": "application/json"
    },
    params={'limit': 15}
)

recent_events = response.json()
```

</RequestExample>
